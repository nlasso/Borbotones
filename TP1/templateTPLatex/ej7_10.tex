y\leftarrow atletasDelPais(j,p), c \leftarrow competenciasAtleta(j,y,p), en(y,sub(ranking(c),3,|ranking(c)-1|))]}

k \leftarrow atletasFracasados(j,p), sgd(y) \geq sgd(k)]}

\asegura{result == masFracasados(j,p)}
\end{problema}
